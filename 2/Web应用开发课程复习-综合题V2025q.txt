1. 编写一个汽车信息的 JavaBean，取名 CarInfoBean。要求如下：
 （1）定义一个包，将该 bean 编译后生成的类存入 com.bean 包中；
 （2）设计三个属性：汽车编号（carNo，字符串型）、 汽车品牌（brand，字符串型）
及是否通过检验（validate，布尔型）；
 （3）设计访问属性的相应方法。
 请按照 JavaBean 的规范编写程序
参考答案：
package com.bean; 
public class CarBean {
 private String carNo;
 private String brand;
 private int year;
 private boolean validate;
 public String getCarNo() { 
 return carNo;
 }
 public void setCarNo(String carNo) {
 this.carNo = carNo;
 }
 public String getBrand() {
 return brand;
 }
 public void setBrand(String brand) {
 this.brand = brand;
 }
 public boolean isValidate() {
 return validate;
 }
 public void setValidate(boolean validate) {
 this.validate = validate;
 }
 public int getYear() {
 return year;
 }

 this.year = year;
 }
}
